    <script>
        function handleLogin() {
            const user = document.getElementById('username').value;
            const pass = document.getElementById('password').value;
            if(user && pass) {
                document.getElementById('display-name').innerText = user;
                document.getElementById('auth-section').classList.add('hidden');
                document.getElementById('tool-section').classList.remove('hidden');
            } else {
                alert("Vui lòng nhập đủ tài khoản và mật khẩu!");
            }
        }

        function soiCau() {
            const btn = document.getElementById('btn-soi');
            const resBox = document.getElementById('result-box');
            const codeBox = document.getElementById('code-val');
            const luckyBox = document.getElementById('lucky-num');

            btn.disabled = true;
            btn.innerText = "ĐANG PHÂN TÍCH...";
            resBox.innerText = "...";
            resBox.className = "text-4xl font-black my-2 text-gray-400 animate-pulse";

            setTimeout(() => {
                const results = ["CÁI", "CON"];
                const finalRes = results[Math.floor(Math.random() * results.length)];
                const code = Math.random().toString(36).substring(2, 10).toUpperCase();
                
                resBox.innerText = finalRes;
                resBox.className = "text-5xl font-black my-2 " + (finalRes === "CÁI" ? "text-red-500" : "text-blue-500");
                codeBox.innerText = code;
                luckyBox.innerText = Math.floor(Math.random() * 10);
                
                btn.disabled = false;
                btn.innerText = "SOI TIẾP";
            }, 1500);
        }
